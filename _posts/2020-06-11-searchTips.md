常用的谷歌搜索技巧
<!--more-->
# Google  
```
site:
# 只搜索某个网站的页面

~XXX
# 搜索与XXX相关的近义词
# ~college  类似于higer education , university

" "
# 双引号一般可以精准匹配

-XXX
# 排除某个关键词
# -zhihu 排除有关zhihu的答案

2010..2020
# 这个时间段的搜索结果

filetype:
# 可以指定搜索类型的文件 
# filetype : pdf 
# filetype : word
# filetype : jpg

intitle:
# 包含关键词

* 
# 通配符 模糊匹配

```
